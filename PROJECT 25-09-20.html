df["FECHAREGISTRO"] = pd.to_datetime(df["FECHAREGISTRO"])
def plot_time( df, time_var, title, rot = 0):
    res = df.groupby([time_var])['SECUENCIARECORRIDO'].count().reset_index(name="count")
    p = res.plot(x = time_var, y = "count", title = title, rot = rot)
    return p
df["hora"] = df["FECHAREGISTRO"].dt.hour
